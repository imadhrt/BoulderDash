create table Projet(
 num int  not null 
 description char(20) not null
 manager int not null
 application int not null 
 budget int not null  constraints budgetcheck check(budget>=10000 and budget<=1000000)
 debut Date default 01/01/2021 not null
 fin Date null  constraints fincheck check(fin>debut)
 constraints numpk primarykey (num)
 
);
